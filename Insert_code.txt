const { Client } = require('@elastic/elasticsearch');
const express = require('express');
const bodyParser = require('body-parser');

const client = new Client({ node: 'http://localhost:9200' }); // Replace with your Elasticsearch server URL
const app = express();

app.use(bodyParser.json());

// Endpoint to update the lessons_learned field
app.post('/update-lessons-learned', async (req, res) => {
  const { issue_id, lessons_learned } = req.body;

  if (!issue_id || !lessons_learned) {
    return res.status(400).send('issue_id and lessons_learned are required');
  }

  try {
    const response = await client.updateByQuery({
      index: 'incidents',
      body: {
        script: {
          source: "ctx._source.lessons_learned = params.lessons_learned",
          lang: "painless",
          params: {
            lessons_learned: lessons_learned
          }
        },
        query: {
          term: { issue_id: issue_id }
        }
      }
    });
    res.json(response);
  } catch (error) {
    console.error('Error updating lessons learned:', error);
    res.status(500).send('Error updating lessons learned');
  }
});

// Start the server
const port = 3000;
app.listen(port, () => {
  console.log(`Server is running on http://localhost:${port}`);
});
